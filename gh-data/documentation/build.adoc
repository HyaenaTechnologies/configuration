== CMake

[,shell]
----
cmake -B ./build

cd ./build

make

./gh-data --help
----

== GNU Make

[,shell]
----
make build

make check

make clean

make format

make run
----

## LLVM Clang

[,shell]
----
# Compile
clang ./src/*/*.h

clang ./src/*.c ./src/*/*.c -include-pch ./src/*/*.pch -o ./build/gh-data

rm ./src/*/*.pch

./build/gh-data

# Analyze
--analyze: Run the static analyzer
-Wall: All Warnings
-Wextra: Extra Warnings
----

== GNU Compiler Collection

[,shell]
----
# Compile
gcc ./src/*/*.h

gcc ./src/*.c ./src/*/*.c -include ./src/*/*.pch -o ./build/gh-data

rm ./src/*/*.pch

./build/gh-data

# Analyze
-fanalyzer
-Wall: All Warnings
-Wextra: Extra Warnings
----

== Install GitHub Data

[,shell]
----
sudo ln -s $PWD/gh-data /usr/local/bin/
----
